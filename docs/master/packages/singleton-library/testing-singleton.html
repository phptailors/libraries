2.4. Testing Singleton

To easily ensure that your class looks like and behaves like a singleton, you may use facilities provided by phptailors/singleton-testing package:

1composer require --dev "phptailors/singleton-testing:dev-master"

Singleton testing is facilitated by AssertIsSingletonTrait. Its basic usage is as simple, as calling the assertIsSingleton() method. The following example demonstrates how one the TrivialSingleton developed in `lib.singleton.trivial_singleton.example`_ can be tested:

 1<?php
 2
 3use Tailors\Lib\Singleton\SingletonTrait;
 4use Tailors\Lib\Singleton\SingletonInterface;
 5use Tailors\Testing\Lib\Singleton\AssertIsSingletonTrait;
 6
 7final class TrivialSingleton implements SingletonInterface
 8{
 9    use SingletonTrait;
10}
11
12final class TrivialSingletonTest extends \PHPUnit\Framework\TestCase
13{
14    use AssertIsSingletonTrait;
15
16    public function testTrivialSingletonClassIsSingleton(): void
17    {
18        $this->assertIsSingleton(TrivialSingleton::class);
19    }
20}
21
22// vim: syntax=php sw=4 ts=4 et:

The following example shows the effect of testing non-singleton classes with the assertIsSingleton() method

  1<?php
  2
  3use Tailors\Testing\Lib\Singleton\AssertIsSingletonTrait;
  4
  5trait PrivateConstructorTrait
  6{
  7    private function __construct()
  8    {
  9    }
 10}
 11
 12trait PublicStaticGetInstanceTrait
 13{
 14    private static $instance = null;
 15
 16    public static function getInstance(): self
 17    {
 18        if (self::$instance === null) {
 19            self::$instance = new self();
 20        }
 21        return self::$instance;
 22    }
 23}
 24
 25trait PrivateCloneTrait
 26{
 27    private function __clone()
 28    {
 29    }
 30}
 31
 32final class NoGetInstance
 33{
 34    use PrivateConstructorTrait;
 35}
 36
 37final class PrivateGetInstance
 38{
 39    use PrivateConstructorTrait;
 40    private static function getInstance(): self
 41    {
 42        return new self();
 43    }
 44}
 45
 46final class NonStaticGetInstance
 47{
 48    use PrivateConstructorTrait;
 49    public function getInstance(): self
 50    {
 51        return new self();
 52    }
 53}
 54
 55final class NoPrivateConstructor
 56{
 57    use PublicStaticGetInstanceTrait;
 58}
 59
 60final class NoPrivateClone
 61{
 62    use PrivateConstructorTrait;
 63    use PublicStaticGetInstanceTrait;
 64}
 65
 66final class NoThrowingWakeup
 67{
 68    use PrivateConstructorTrait;
 69    use PublicStaticGetInstanceTrait;
 70    use PrivateCloneTrait;
 71}
 72
 73final class ThrowingWrongExceptionWakeup
 74{
 75    use PrivateConstructorTrait;
 76    use PublicStaticGetInstanceTrait;
 77    use PrivateCloneTrait;
 78
 79    public function __wakeup()
 80    {
 81        throw new \RuntimeException();
 82    }
 83}
 84
 85final class NonSingletonTest extends \PHPUnit\Framework\TestCase
 86{
 87    use AssertIsSingletonTrait;
 88
 89    public function testNoGetInstanceIsSingleton(): void
 90    {
 91        $this->assertIsSingleton(NoGetInstance::class);
 92    }
 93
 94    public function testPrivateGetInstanceIsSingleton(): void
 95    {
 96        $this->assertIsSingleton(PrivateGetInstance::class);
 97    }
 98
 99    public function testNonStaticGetInstanceIsSingleton(): void
100    {
101        $this->assertIsSingleton(NonStaticGetInstance::class);
102    }
103
104    public function testNoPrivateConstructorIsSingleton(): void
105    {
106        $this->assertIsSingleton(NoPrivateConstructor::class);
107    }
108
109    public function testNoPrivateCloneIsSingleton(): void
110    {
111        $this->assertIsSingleton(NoPrivateClone::class);
112    }
113
114    public function testNoThrowingWakeupIsSingleton(): void
115    {
116        $this->assertIsSingleton(NoThrowingWakeup::class);
117    }
118
119    public function testThrowingWrongExceptionWakeupIsSingleton(): void
120    {
121        $this->assertIsSingleton(ThrowingWrongExceptionWakeup::class);
122    }
123}
